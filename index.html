3 - Distância entre Planos


A1 = 2
B1 = -3
C1 = 4
D1 = -5
A2 = 2
B2 = -3
C2 = 4
D2 = 7

if (A1 * B2 == A2 * B1) and (A1 * C2 == A2 * C1) and (B1 * C2 == B2 * C1):
    numerador = abs(D1 - D2)
    denominador = (A1**2 + B1**2 + C1**2) ** 0.5
    distancia = numerador / denominador
    print("Distância entre os planos:", round(distancia, 2))
